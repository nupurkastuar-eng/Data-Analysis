# Filtering the nouns and adjectives and Getting the product ID

import re
import nltk
from nltk.tokenize import word_tokenize
from nltk.corpus import stopwords
from nltk.stem import WordNetLemmatizer
from nltk.tag import pos_tag


# Extract product ID from product URL
def extract_product_id(url):
    """
    Amazon review URLs look like:
    https://www.amazon.com/product-reviews/B00004O...
    Product ID starts with 'B' and is 10 characters long.
    """
    match = re.search(r"/([A-Z0-9]{10})(?=[/?])", url)
    if match:
        return match.group(1)
    return None


# Filter only nouns + adjectives ---
lemmatizer = WordNetLemmatizer()

def clean_nouns_adjectives(text):

    # Lowercase
    text = text.lower()

    # Remove special characters
    text = re.sub(r"[^a-zA-Z0-9\s]", " ", text)

    # Tokenize
    tokens = word_tokenize(text)

    # Remove stopwords
    stop_words = set(stopwords.words("english"))
    filtered_tokens = [w for w in tokens if w not in stop_words]

    # POS tagging
    tagged_words = pos_tag(filtered_tokens)

    # Extract nouns & adjectives
    nouns = [lemmatizer.lemmatize(w, "n") for w, t in tagged_words if t.startswith("NN")]
    adjs  = [lemmatizer.lemmatize(w, "a") for w, t in tagged_words if t.startswith("JJ")]

    # Combine nouns + adjectives
    combined = nouns + adjs

    # Return as single string
    return " ".join(combined)

# Create a dictionary and corpus for LDA
dictionary = Dictionary(token_list)
corpus = [dictionary.doc2bow(text) for text in token_list]

coherence_scores = []

# Train LDA models for topics 210
for k in range(2, 11):
    lda_model = LdaModel(
        corpus=corpus,
        id2word=dictionary,
        num_topics=k,
        random_state=0,   # fixing random state for reproducibility
        passes=10,        # number of passes through the corpus
        iterations=100    # number of iterations per pass
    )

    # Calculate coherence score using c_v method
    coherence_model = CoherenceModel(
        model=lda_model,
        texts=token_list,
        dictionary=dictionary,
        coherence='c_v'
    )

    coherence = coherence_model.get_coherence()
    coherence_scores.append((k, coherence))

# Select the number of topics with highest coherence
best_k = max(coherence_scores, key=lambda x: x[1])[0]
